00048     // find the most significant bit smaller than start by which code0 and code1 differ
00049     template <typename Integer>
00050     FORCE_INLINE NIH_HOST_DEVICE int32 find_leading_bit_difference(
00051         const  int32  start_level,
00052         const Integer code0,
00053         const Integer code1)
00054     {
00055         int32 level = start_level;
00056 
00057         while (level >= 0)
00058         {
00059             const Integer mask = Integer(1u) << level;
00060 
00061             if ((code0 & mask) !=
00062                 (code1 & mask))
00063                 break;
00064 
00065             --level;
00066         }
00067         return level;
00068     }

00100     while (task_queues[ in_queue ].size())
00101     {
00102         task_queues[ out_queue ].erase(
00103             task_queues[ out_queue ].begin(),
00104             task_queues[ out_queue ].end() );
00105             
00106         for (uint32 task_id = 0; task_id < task_queues[ in_queue ].size(); ++task_id)
00107         {
00108             const Split_task task  = task_queues[ in_queue ][ task_id ];
00109             const uint32 skip_node = skip_queues[ in_queue ][ task_id ];

00111             const uint32 node  = task.m_node;
00112             const uint32 begin = task.m_begin;
00113             const uint32 end   = task.m_end;
00114                   uint32 level = task.m_input;
00115 
00116             if (!keep_singletons)
00117             {
00118                 level = bintree::find_leading_bit_difference(
00119                     level,
00120                     codes[ begin ],
00121                     codes[ end-1 ] );
00122             }
00123 
00124             uint32 output_count = 0;
00125             uint32 split_index;
00126 
00127             // check whether the input node really needs to be split
00128             if (end - begin > max_leaf_size && level != uint32(-1))
00129             {
00130                 // find the "partitioning pivot" using a binary search
00131                 split_index = find_pivot(
00132                     codes + begin,
00133                     end - begin,
00134                     mask_and<Integer>( Integer(1u) << level ) ) - codes;
00135 
00136                 output_count = (split_index == begin || split_index == end) ? 1u : 2u;
00137             }
00138 
00139             const uint32 node_offset = node_count;
00140             const uint32 first_end   = (output_count == 1) ? end       : split_index;
00141             const uint32 first_skip  = (output_count == 1) ? skip_node : node_offset+1;
00142 
00143             if (output_count >= 1) { task_queues[ out_queue ].push_back( Split_task( node_offset+0, begin, first_end, level-1 ) ); skip_queues[ out_queue ].push_back( first_skip ); }

00145 
00146             const bool generate_leaf = output_count == 0;
00147 
00148             // count how many leaves we need to generate
00149             const uint32 leaf_index = leaf_count;
00150 
00151             node_count += output_count;
00152             leaf_count += generate_leaf ? 1 : 0;
00153 
00154             // write the parent node
00155             context.write_node(
00156                 node,
00157                 output_count ? split_index != begin : false,
00158                 output_count ? split_index != end   : false,
00159                 output_count ? node_offset          : leaf_index,
00160                 skip_node );
00161 
00162             // make a leaf if necessary
00163             if (generate_leaf)
00164                 context.write_leaf( leaf_index, begin, end );
00165         }
00166 
00167         std::swap( in_queue, out_queue );
00168     }
