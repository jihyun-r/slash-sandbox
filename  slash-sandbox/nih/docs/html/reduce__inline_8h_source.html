00039 template <uint32 BLOCK_SIZE, typename Tree, typename Input_iterator, typename Output_iterator, typename Operator>
00040 __global__ void reduce_leaves_kernel(
00041     const Tree              tree,
00042     const uint32            n_leaves,
00043     const Input_iterator    in_values,
00044     Output_iterator         out_values,
00045     const Operator          op,
00046     const typename std::iterator_traits<Output_iterator>::value_type def_value)
00047 {
00048     const uint32 grid_size = gridDim.x * blockDim.x;
00049 
00050     // loop through all logical blocks associated to this physical one
00051     for (uint32 base_idx = blockIdx.x * BLOCK_SIZE;
00052         base_idx < n_leaves;
00053         base_idx += grid_size)
00054     {
00055         const uint32 leaf_id = threadIdx.x + base_idx;
00056 
00057         if (leaf_id < n_leaves)
00058         {
00059             const uint2 leaf = tree.get_leaf( leaf_id );
00060             const uint32 begin = leaf.x;
00061             const uint32 end   = leaf.y;
00062 
00063             typename std::iterator_traits<Output_iterator>::value_type value = def_value;
00064             for (uint32 i = begin; i < end; ++i)
00065                 value = op( value, thrust::detail::device::dereference( in_values + i ) );
00066 
00067             thrust::detail::device::dereference( out_values, leaf_id ) = value;
00068         }
00069     }

00099 template <uint32 BLOCK_SIZE, typename Tree, typename Leaf_iterator, typename Output_iterator, typename Operator>
00100 __global__ void reduce_level_kernel(
00101     const Tree              tree,
00102     const uint32            begin,
00103     const uint32            end,
00104     Leaf_iterator           leaf_values,
00105     Output_iterator         out_values,
00106     const Operator          op)
00107 {
00108     const uint32 grid_size = gridDim.x * blockDim.x;
00109 
00110     // loop through all logical blocks associated to this physical one
00111     for (uint32 base_idx = blockIdx.x * BLOCK_SIZE + begin;
00112         base_idx < end;
00113         base_idx += grid_size)
00114     {
00115         const uint32 node_id = threadIdx.x + base_idx;
00116 
00117         if (node_id < end)
00118         {
00119             const typename Tree::node_type node = tree.get_node( node_id );
00120 
00121             if (node.is_leaf())
00122             {
00123                 // copy the corresponding leaf value
00124                 const uint32 leaf_index = node.get_leaf_index();
00125                 thrust::detail::device::dereference( out_values + node_id ) = thrust::detail::device::dereference( leaf_values + leaf_index );
00126             }
00127             else
00128             {
00129                 // reduce all child values
00130                 const uint32 n_children = node.get_child_count();
00131 
00132                 typename std::iterator_traits<Output_iterator>::value_type value =
00133                     thrust::detail::device::dereference( out_values, node.get_child(0) );
00134                 for (uint32 i = 1; i < n_children; ++i)
00135                     value = op( value, thrust::detail::device::dereference( out_values, node.get_child(i) ) );
00136 
00137                 thrust::detail::device::dereference( out_values, node_id ) = value;
00138             }
00139         }
00140     }

00192             tree,
00193             in_values,
00194             leaf_values,
00195             op,
00196             def_value );
00197 
00198         const uint32 n_levels = tree.get_level_count();
00199 
00200         uint32 offset = 0;
00201 
00202         for (int32 level = n_levels-1; level >= 0; --level)
00203         {
00204             const uint32 level_begin = tree.get_level(level);
00205             const uint32 level_end   = tree.get_level(level+1);
00206             const uint32 level_size  = level_end - level_begin;
00207 
00208             if (level_size == 0)
00209                 continue;
